#!/bin/bash

# Define server details (replace with your actual information)
HOST="sftp_server_hostname"
PORT=22  # default SFTP port
USERNAME="your_username"
PASSWORD="your_password"

# Define remote and local directories (replace with your actual paths)
REMOTE_DIR="/path/to/remote/install"
LOCAL_DIR="/path/to/local/local_installer"

# Download all files from the remote directory
for file in $(curl -sftp sftp://$USERNAME:$PASSWORD@$HOST:$PORT -l $REMOTE_DIR); do
  # Extract filename from listing output
  filename=$(basename "$file")
  curl -sftp sftp://$USERNAME:$PASSWORD@$HOST:$PORT -O $LOCAL_DIR/$filename $REMOTE_DIR/$filename
done

echo "Download completed."

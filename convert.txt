import yaml

# Original YAML data
data = """
key1:
  - value1
  - value2
key2:
  - value3
  - value4
"""

# Load the YAML data into a Python dictionary
yaml_data = yaml.safe_load(data)

# Convert the lists to the desired format
for key in yaml_data:
    yaml_data[key] = str(yaml_data[key])

# Convert back to YAML format
formatted_yaml = yaml.dump(yaml_data, default_flow_style=True)

print(formatted_yaml)

# Original data as a string
data = """
key1:
  - value1
  - value2
key2:
  - value3
  - value4
"""

# Parsing the data manually
result = {}
lines = data.strip().splitlines()
current_key = None

for line in lines:
    # Check if the line represents a key
    if not line.startswith(" "):
        current_key = line.rstrip(":")
        result[current_key] = []
    # Otherwise, it's a value for the current key
    else:
        value = line.strip(" -")
        result[current_key].append(value)

# Format the output to the desired representation using string formatting
formatted_result = {k: '["{}"]'.format('", "'.join(vals)) for k, vals in result.items()}

# Print the formatted result
for key, value in formatted_result.items():
    print(f"{key}: {value}")
